-- HTTP kompatibel semua executor (Synapse, Fluxus, Delta, Arceus, dll)
local function httpGet(url)
    local body = nil
    -- Method 1: syn.request (Synapse X)
    if not body then pcall(function()
        if syn and syn.request then
            local r = syn.request({Url=url,Method="GET"})
            if r and r.StatusCode==200 then body=r.Body end
        end
    end) end
    -- Method 2: request() global (banyak executor)
    if not body then pcall(function()
        if request then
            local r = request({Url=url,Method="GET"})
            if r and r.StatusCode==200 then body=r.Body end
        end
    end) end
    -- Method 3: http.request
    if not body then pcall(function()
        if http and http.request then
            local r = http.request({Url=url,Method="GET"})
            if r and r.StatusCode==200 then body=r.Body end
        end
    end) end
    -- Method 4: fluxus/executor request
    if not body then pcall(function()
        if fluxus and fluxus.request then
            local r = fluxus.request({Url=url,Method="GET"})
            if r and r.StatusCode==200 then body=r.Body end
        end
    end) end
    -- Method 5: game:HttpGet
    if not body then pcall(function()
        body = game:HttpGet(url, true)
    end) end
    -- Method 6: HttpService (fallback)
    if not body then pcall(function()
        body = game:GetService("HttpService"):GetAsync(url, true)
    end) end
    return body
end

local function urlEnc(s)
    return tostring(s):gsub("[^%w%-_%.~]", function(c)
        return ("%%%02X"):format(c:byte())
    end)
end

local function getHWID()
    local h = ""
    pcall(function() if gethwid then h = tostring(gethwid()) end end)
    if h == "" then
        pcall(function()
            h = tostring(game:GetService("Players").LocalPlayer.UserId).."_RBX"
        end)
    end
    return h ~= "" and h or "no_hwid"
end

local function verifyKey(key)
    local hs = game:GetService("HttpService")
    local pl = game:GetService("Players").LocalPlayer
    local url = KEY_API.."?key="..urlEnc(key).."&user="..urlEnc(pl.Name).."&hwid="..urlEnc(getHWID())
    local raw = httpGet(url)
    if not raw then
        return false, "⚠ HTTP gagal. Cek executor support HttpGet/request()"
    end
    local ok2, data = pcall(function() return hs:JSONDecode(raw) end)
    if not ok2 or type(data)~="table" then
        return false, "⚠ Response error dari server"
    end
    return data.success==true, data.message or "Error", data
end

local function serializeData(data)
    local t={}
    for _,d in ipairs(data) do
        table.insert(t,{
            px=d.Position.X, py=d.Position.Y, pz=d.Position.Z,
            cfx=d.CFrame.X, cfy=d.CFrame.Y, cfz=d.CFrame.Z,
            time=d.Time, state=d.State,
            vx=d.Velocity and d.Velocity.X or 0,
            vy=d.Velocity and d.Velocity.Y or 0,
            vz=d.Velocity and d.Velocity.Z or 0,
            ws=d.WalkSpeed or 27,
        })
    end
    return HttpService:JSONEncode(t)
end

local function deserializeData(json)
    local t=HttpService:JSONDecode(json)
    local data={}
    for _,d in ipairs(t) do
        table.insert(data,{
            Position  = Vector3.new(d.px,d.py,d.pz),
            CFrame    = CFrame.new(d.cfx or d.px, d.cfy or d.py, d.cfz or d.pz),
            Time      = d.time, State=d.state,
            Velocity  = Vector3.new(d.vx or 0,d.vy or 0,d.vz or 0),
            WalkSpeed = d.ws or 27,
        })
    end
    return data
end

local SAVE_FOLDER = "SanzxmzzWalk/"
local function ensureSaveFolder()
    pcall(function() if not isfolder(SAVE_FOLDER) then makefolder(SAVE_FOLDER) end end)
end
local function saveToFile(name, data)
    local ok,json = pcall(serializeData, data)
    if not ok then return false end
    -- Pastikan folder ada
    pcall(function()
        if not isfolder(SAVE_FOLDER) then makefolder(SAVE_FOLDER) end
    end)
    -- Coba berbagai path
    local saved=false
    for _,path in ipairs({SAVE_FOLDER..name..".json", "./"..SAVE_FOLDER..name..".json", name..".json"}) do
        local ok2=pcall(function() writefile(path, json) end)
        if ok2 then saved=true; break end
    end
    return saved
end
local function loadFromFile(name)
    -- Coba berbagai path format
    local content=nil
    local tryPaths={
        SAVE_FOLDER..name..".json",
        SAVE_FOLDER..name,
        "./"..SAVE_FOLDER..name..".json",
        name..".json",
    }
    for _,path in ipairs(tryPaths) do
        local ok,res=pcall(function() return readfile(path) end)
        if ok and res and #res>2 then content=res; break end
    end
    if not content then return nil end
    local ok2,data = pcall(deserializeData, content)
    return ok2 and data or nil
end
local function listSaveFiles()
    local files={}
    -- Coba berbagai path format yang berbeda-beda di tiap executor
    local tryFolders={SAVE_FOLDER, SAVE_FOLDER:gsub("/$",""), "./"..SAVE_FOLDER, SAVE_FOLDER:lower()}
    local found=false
    for _,folder in ipairs(tryFolders) do
        local ok,result=pcall(function()
            if not isfolder(folder) then makefolder(folder) end
            local lst=listfiles(folder)
            if lst and #lst>=0 then
                for _,f in ipairs(lst) do
                    local name=f:match("([^/\\]+)%.json$") or f:match("([^/\\]+)$")
                    if name then name=name:gsub("%.json$",""); table.insert(files,name) end
                end
                found=true
            end
        end)
        if found and #files>=0 then break end
    end
    -- Jika listfiles tidak work, coba readdir (beberapa executor pakai ini)
    if not found then
        pcall(function()
            local lst=readdir and readdir(SAVE_FOLDER) or {}
            for _,f in ipairs(lst) do
                local name=f:match("([^/\\]+)%.json$") or f:match("([^/\\]+)$")
                if name then name=name:gsub("%.json$",""); table.insert(files,name) end
            end
        end)
    end
    table.sort(files)
    return files
end

local function notify(title, body, dur)
    local t=(title or ""):lower()
    local ntype="info"
    if t:find("⬤") or t:find("rec") then ntype="rec"
    elseif t:find("▶") or t:find("play") then ntype="play"
    elseif t:find("⏹") or t:find("stop") or t:find("🗑") or t:find("clear") then ntype="error"
    elseif t:find("💾") or t:find("saved") or t:find("✅") or t:find("load") then ntype="success"
    elseif t:find("⚠") or t:find("warn") or t:find("error") then ntype="warning"
    end
    showNotif(title, body or "", ntype, dur or 3)
end

local activeDropdown=nil
local function mkDropdown(labelTxt,options,defaultTxt,par,onChange)
    local currentVal=defaultTxt   -- ← STATE INTERNAL, tidak tergantung btn.Text
    local wrapper=Instance.new("Frame"); wrapper.Size=UDim2.new(1,0,0,28); wrapper.BackgroundTransparency=1; wrapper.Parent=par
    local lbl=Instance.new("TextLabel"); lbl.Size=UDim2.new(0.40,-4,1,0); lbl.BackgroundTransparency=1; lbl.Text=labelTxt
    lbl.TextColor3=Color3.fromRGB(190,200,220); lbl.Font=Enum.Font.Gotham; lbl.TextSize=9; lbl.TextXAlignment=Enum.TextXAlignment.Left; lbl.Parent=wrapper
    local btn=Instance.new("TextButton"); btn.Size=UDim2.new(0.60,0,1,0); btn.Position=UDim2.new(0.40,0,0,0)
    btn.BackgroundColor3=Color3.fromRGB(12,22,44); btn.Text=defaultTxt.." ▾"; btn.TextColor3=Color3.fromRGB(0,200,255)
    btn.Font=Enum.Font.GothamSemibold; btn.TextSize=10; btn.BorderSizePixel=0; btn.Parent=wrapper
    Instance.new("UICorner",btn).CornerRadius=UDim.new(0,6)
    do local sk=Instance.new("UIStroke"); sk.Color=Color3.fromRGB(0,80,150); sk.Thickness=1; sk.Parent=btn end
    btn.MouseButton1Click:Connect(function()
        if activeDropdown then activeDropdown:Destroy(); activeDropdown=nil end
        local ITEM_H=24; local MAX_V=7; local totalH=#options*ITEM_H; local visH=math.min(totalH,MAX_V*ITEM_H)
        local popup=Instance.new("Frame"); popup.Size=UDim2.new(0,155,0,visH+6)
        popup.BackgroundColor3=Color3.fromRGB(8,16,36); popup.BorderSizePixel=0; popup.ZIndex=200
        Instance.new("UICorner",popup).CornerRadius=UDim.new(0,8)
        do local sk=Instance.new("UIStroke"); sk.Color=Color3.fromRGB(0,100,200); sk.Thickness=1; sk.Parent=popup end
        local sf=Instance.new("ScrollingFrame"); sf.Size=UDim2.new(1,0,1,0); sf.BackgroundTransparency=1
        sf.ScrollBarThickness=3; sf.ScrollBarImageColor3=Color3.fromRGB(0,140,255)
        sf.CanvasSize=UDim2.new(0,0,0,totalH+6); sf.BorderSizePixel=0; sf.ZIndex=200; sf.Parent=popup
        local ul=Instance.new("UIListLayout"); ul.Padding=UDim.new(0,1); ul.SortOrder=Enum.SortOrder.LayoutOrder; ul.Parent=sf
        local pad2=Instance.new("UIPadding"); pad2.PaddingTop=UDim.new(0,3); pad2.PaddingLeft=UDim.new(0,3); pad2.PaddingRight=UDim.new(0,3); pad2.Parent=sf
        for i,opt in ipairs(options) do
            local isSel=(opt==currentVal)
            local ob=Instance.new("TextButton"); ob.Size=UDim2.new(1,0,0,ITEM_H-2); ob.LayoutOrder=i
            ob.BackgroundColor3=isSel and Color3.fromRGB(0,60,130) or Color3.fromRGB(14,26,52)
            ob.Text=tostring(opt); ob.TextColor3=isSel and Color3.fromRGB(0,230,255) or Color3.fromRGB(200,215,240)
            ob.Font=Enum.Font.Gotham; ob.TextSize=10; ob.BorderSizePixel=0; ob.ZIndex=200; ob.Parent=sf
            Instance.new("UICorner",ob).CornerRadius=UDim.new(0,5)
            ob.MouseButton1Click:Connect(function()
                currentVal=opt          -- UPDATE STATE INTERNAL
                btn.Text=tostring(opt).." ▾"
                onChange(opt)
                popup:Destroy(); activeDropdown=nil
            end)
            ob.MouseEnter:Connect(function() if not isSel then ob.BackgroundColor3=Color3.fromRGB(0,45,90) end end)
            ob.MouseLeave:Connect(function() if not isSel then ob.BackgroundColor3=Color3.fromRGB(14,26,52) end end)
        end
        local absY=btn.AbsolutePosition.Y+btn.AbsoluteSize.Y+4; local absX=btn.AbsolutePosition.X
        if absY+visH+6>screenGui.AbsoluteSize.Y then absY=btn.AbsolutePosition.Y-(visH+6)-4 end
        popup.Position=UDim2.new(0,absX,0,absY); popup.Parent=screenGui; activeDropdown=popup
        local cc; cc=UserInputService.InputBegan:Connect(function(inp)
            if inp.UserInputType==Enum.UserInputType.MouseButton1 or inp.UserInputType==Enum.UserInputType.Touch then
                task.delay(0.15,function()
                    if activeDropdown~=popup then cc:Disconnect(); return end
                    local mp=UserInputService:GetMouseLocation(); local pp=popup.AbsolutePosition; local ps=popup.AbsoluteSize
                    if not(mp.X>=pp.X and mp.X<=pp.X+ps.X and mp.Y>=pp.Y and mp.Y<=pp.Y+ps.Y) then
                        if activeDropdown==popup then popup:Destroy(); activeDropdown=nil end
                        cc:Disconnect()
                    end
                end)
            end
        end)    end)
    return wrapper,btn
end

local function refreshKeyInfo()
    if not _verifyData or not _verifyData.success then return end
    pcall(function()
        -- Username
        kiUsername.Text="👤 Username  : "..(_verifyData.username or "—")

        -- Format expired_at dari ISO string jadi readable
        local expStr = "—"
        if _verifyData.expires_at then
            -- ISO: "2025-01-30T12:00:00.000Z" → "30 Jan 2025, 12:00"
            local y,mo,d,h,mi = tostring(_verifyData.expires_at):match("(%d+)-(%d+)-(%d+)T(%d+):(%d+)")
            local months={"Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"}
            if y then
                local mName = months[tonumber(mo)] or mo
                expStr = d.." "..mName.." "..y..", "..h..":"..mi
                -- Cek apakah sudah expired
                local nowStr = os.date("!%Y-%m-%dT%H:%M:%S")
                if _verifyData.expires_at < nowStr then
                    expStr = expStr.." (EXPIRED)"
                end
            else
                expStr = tostring(_verifyData.expires_at)
            end
        elseif _verifyData.duration_type == "lifetime" then
            expStr = "∞ Lifetime (Tidak Expired)"
        end
        kiExpired.Text="⏳ Expired Key : "..expStr

        -- Tier dari duration_type
        local dur = _verifyData.duration_type or "24h"
        local tierStr
        if dur == "lifetime" then tierStr = "LIFETIME"
        elseif dur == "30d" or dur == "60d" then tierStr = "PREMIUM"
        elseif dur == "7d" or dur == "5d" then tierStr = "STANDARD"
        else tierStr = "FREE ("..dur..")" end
        kiTier.Text="👑 Tier         : "..tierStr
        local tierColor = (dur=="lifetime") and Color3.fromRGB(255,214,10)
            or (dur=="30d" or dur=="60d") and Color3.fromRGB(180,120,255)
            or Color3.fromRGB(0,200,255)
        kiTier.TextColor3=tierColor
    end)
end

local managedTracks={}; local currentPlayingState=nil
local VALID_STATES={Walk=true,Run=true,Idle=true,Jump=true,Fall=true,Climb=true}

local function setAnimateEnabled(on)
    pcall(function()
        local a=player.Character and player.Character:FindFirstChild("Animate")
        if a then a.Disabled=not on end
    end)
end
local function loadTracksFromAnimate()
    for _,t in pairs(managedTracks) do pcall(function() t:Stop(0) end); pcall(function() t:Destroy() end) end
    managedTracks={}; currentPlayingState=nil
    pcall(function()
        local c=player.Character; if not c then return end
        local hm=c:FindFirstChildOfClass("Humanoid"); local am=hm and hm:FindFirstChildOfClass("Animator")
        local aS=c:FindFirstChild("Animate"); if not am or not aS then return end
        local fm={walk="Walk",run="Run",idle="Idle",jump="Jump",fall="Fall",climb="Climb"}
        for _,folder in ipairs(aS:GetChildren()) do
            local st=fm[folder.Name:lower()]
            if st and not managedTracks[st] then
                for _,ch in ipairs(folder:GetChildren()) do
                    if ch:IsA("Animation") and ch.AnimationId~="" then
                        local fa=Instance.new("Animation"); fa.AnimationId=ch.AnimationId
                        local ok2,tr=pcall(function() return am:LoadAnimation(fa) end); fa:Destroy()
                        if ok2 and tr then tr.Priority=Enum.AnimationPriority.Action3; managedTracks[st]=tr; break end
                    end
                end
            end
        end
    end)
end
local function stopAllManagedAnims(fade)
    for _,t in pairs(managedTracks) do if t and t.IsPlaying then pcall(function() t:Stop(fade or 0.15) end) end end
    currentPlayingState=nil
end
local function driveAnim(st)
    if currentPlayingState==st then return end
    for s,t in pairs(managedTracks) do if t and t.IsPlaying and s~=st then t:Stop(0.15) end end
    local tgt=managedTracks[st]; if tgt and not tgt.IsPlaying then tgt:Play(0.15) end
    currentPlayingState=st
end

local activeBodyVel=nil
local function removeBodyVel() if activeBodyVel and activeBodyVel.Parent then activeBodyVel:Destroy() end; activeBodyVel=nil end
local function applyBodyVelocity(vel,dur)
    removeBodyVel(); if not rootPart then return end
    local bv=Instance.new("BodyVelocity"); bv.Velocity=vel; bv.MaxForce=Vector3.new(8000,8000,8000); bv.P=4000; bv.Parent=rootPart

local prevPBState=nil; local airFrames=0
local function applyPlaybackState(st,vel,ws)
    if not humanoid then return end
    if not VALID_STATES[st] then st="Idle" end
    local wasAir=(prevPBState=="Jump" or prevPBState=="Fall")
    local nowGnd=(st=="Walk" or st=="Run" or st=="Idle" or st=="Climb")
    if wasAir and nowGnd then
        airFrames=0; removeBodyVel()
        pcall(function() rootPart.AssemblyLinearVelocity=Vector3.zero; rootPart.AssemblyAngularVelocity=Vector3.zero end)
        humanoid:ChangeState(Enum.HumanoidStateType.Landed)
        task.defer(function() if humanoid then humanoid:ChangeState(Enum.HumanoidStateType.Running) end end)
    end
    if st=="Jump" or st=="Fall" then airFrames=airFrames+1 else airFrames=0 end
    prevPBState=st
    if st=="Jump" then
        humanoid:ChangeState(Enum.HumanoidStateType.Jumping); driveAnim("Jump")
        if airFrames<=3 and vel and vel.Y>0.5 then
            local fac=math.max(0.1,1-(airFrames-1)*0.3)
            applyBodyVelocity(Vector3.new(vel.X*0.12,vel.Y*0.45*fac,vel.Z*0.12),0.06)
        end
    elseif st=="Fall" then humanoid:ChangeState(Enum.HumanoidStateType.Freefall); driveAnim("Fall")
    elseif st=="Climb" then removeBodyVel(); humanoid:ChangeState(Enum.HumanoidStateType.Climbing); driveAnim("Climb")
    elseif st=="Run" then removeBodyVel(); humanoid.WalkSpeed=ws or 27; humanoid:ChangeState(Enum.HumanoidStateType.Running); driveAnim("Run")
    elseif st=="Walk" then removeBodyVel(); humanoid.WalkSpeed=ws or 25; humanoid:ChangeState(Enum.HumanoidStateType.Running); driveAnim("Walk")
    else removeBodyVel(); humanoid:ChangeState(Enum.HumanoidStateType.Running); driveAnim("Idle") end
end

local function getHumanoidState()
    if not humanoid or not rootPart then return "Idle" end
    local st=humanoid:GetState(); local spd=humanoid.MoveDirection.Magnitude; local vy=rootPart.AssemblyLinearVelocity.Y
    if st==Enum.HumanoidStateType.Jumping then return "Jump"
    elseif st==Enum.HumanoidStateType.Freefall then return vy>1 and "Jump" or "Fall"
    elseif st==Enum.HumanoidStateType.Climbing then return "Climb"
    elseif spd>0.1 then return humanoid.WalkSpeed>=20 and "Run" or "Walk"
    else return "Idle" end
end

local function refreshPlayUI()
    if isPlaying then
        local txt=autoPlayTarget>0 and string.format("▶ %d poin | Auto %d/%d",#recordedData,autoPlayCount,autoPlayTarget) or "▶ Playing | "..#recordedData.." poin"
        playStatusLbl.Text=txt; playStatusLbl.TextColor3=Color3.fromRGB(80,255,100)
    elseif isWalkingToTrack then
        playStatusLbl.Text="🏃 Menuju track terdekat..."; playStatusLbl.TextColor3=Color3.fromRGB(255,200,0)
    else
        playStatusLbl.Text="● Idle | "..#recordedData.." poin"; playStatusLbl.TextColor3=Color3.fromRGB(100,170,100)
    end
end

local function stopPlayback()
    if not isPlaying then return end
    if playbackConn then playbackConn:Disconnect(); playbackConn=nil end
    isPlaying=false; isPaused=false; stopAllManagedAnims(); removeBodyVel()
    prevPBState=nil; airFrames=0; setAnimateEnabled(true)
    updateStatus("Playback Stopped","Ready",Color3.fromRGB(200,200,220),Color3.fromRGB(80,80,80))
    animLbl.Text="State: — | —"; notify("⏹ Stopped","Playback selesai",2); updateStats(); refreshPlayUI()
end

local function findNearestIndex()
    if #recordedData==0 then return 1 end
    local ch=player.Character; local rp=ch and ch:FindFirstChild("HumanoidRootPart")
    if not rp then return 1 end
    local myPos=rp.Position; local bestIdx=1; local bestDist=math.huge
    for i,d in ipairs(recordedData) do local dist=(d.Position-myPos).Magnitude; if dist<bestDist then bestDist=dist; bestIdx=i end end
    return bestIdx
end

local _pbUISkip=0
local function startPlayback(fromNearest)
    if isPlaying then return end
    if #recordedData<2 then notify("⚠ Error","Butuh ≥ 2 titik",2); return end
    character=player.Character; if not character then return end
    humanoid=character:FindFirstChildOfClass("Humanoid"); rootPart=character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not rootPart then return end
    local startIdx=fromNearest and findNearestIndex() or 1
    if not fromNearest then pcall(function() rootPart.CFrame=recordedData[1].CFrame end) end
    updateStatus(fromNearest and "▶ Playing (Terdekat)..." or "▶ Playing (Awal)...",fromNearest and "Dari titik terdekat" or "Dari awal",Color3.fromRGB(80,255,140),Color3.fromRGB(0,220,100))
    setAnimateEnabled(false); loadTracksFromAnimate(); isPlaying=true; isPaused=false
    notify("▶ Playing","LShift untuk stop",2)
    -- FIX NEAREST: offset elapsed ke startIdx agar playback mulai dari titik terdekat, bukan mundur
    local elapsed=0
    for i=1,startIdx-1 do elapsed=elapsed+(recordedData[i].Time or 0.05) end
    local lastTick=tick(); local delayLeft=playbackDelay
    playbackConn=RunService.Heartbeat:Connect(function()
        if not isPlaying then return end
        pcall(function()
        if not character or not character.Parent then stopPlayback(); return end
        if isPaused then lastTick=tick(); return end
        if delayLeft>0 then local n2=tick(); delayLeft=delayLeft-(n2-lastTick); lastTick=n2; animLbl.Text=string.format("⏱ %.1fs...",math.max(0,delayLeft)); return end
        local now=tick(); elapsed=elapsed+(now-lastTick)*playbackSpeed; lastTick=now
        local seg,segE=nil,elapsed
        for i=1,#recordedData-1 do
            local dur=recordedData[i].Time or 0.05
            if segE<=dur then seg=i; break end; segE=segE-dur
        end
        if not seg then
            if loopPlayback or (autoPlayTarget>0 and autoPlayCount<autoPlayTarget) then
                if autoPlayTarget>0 then
                    autoPlayCount=autoPlayCount+1; updateStats()
                    if autoPlayCount>=autoPlayTarget then autoPlayCount=0; autoPlayTarget=0; stopPlayback(); return end
                end
                -- FIX AUTOPLAY LOOP: reset posisi + velocity + tunggu physics
                elapsed=0; lastTick=tick()
                pcall(function()
                    rootPart.AssemblyLinearVelocity=Vector3.zero
                    rootPart.AssemblyAngularVelocity=Vector3.zero
                    task.wait(0.04)
                    rootPart.CFrame=recordedData[1].CFrame
                    task.wait(0.06) -- tunggu physics settle
                    rootPart.AssemblyLinearVelocity=Vector3.zero
                    lastTick=tick() -- reset timer setelah settle
                end)
            else stopPlayback() end
            return
        end
        local cur=recordedData[seg]; local nxt=recordedData[seg+1]
        local t=math.clamp(segE/(cur.Time or 0.05),0,1)
        pcall(function()
            if smoothMovement then
                local newPos=cur.Position:Lerp(nxt.Position,t); local lv=cur.CFrame.LookVector:Lerp(nxt.CFrame.LookVector,t)
                if lv.Magnitude>0.01 then rootPart.CFrame=CFrame.lookAt(newPos,newPos+lv) else rootPart.CFrame=CFrame.new(newPos) end
            else rootPart.CFrame=(t>=0.5) and nxt.CFrame or cur.CFrame end
        end)
        local st=(t>=0.5) and nxt.State or cur.State
        pcall(function() if st~="Jump" and st~="Fall" then rootPart.AssemblyLinearVelocity=Vector3.zero; rootPart.AssemblyAngularVelocity=Vector3.zero end end)
        local vel=(t>=0.5) and nxt.Velocity or cur.Velocity; local ws=(t>=0.5) and (nxt.WalkSpeed or 25) or (cur.WalkSpeed or 26)
        applyPlaybackState(st,vel,ws)
        _pbUISkip=_pbUISkip+1
        if _pbUISkip>=3 then
            _pbUISkip=0
            animLbl.Text=string.format("▶ %d/%d  🏃%s  ⚡%.1fx",seg,#recordedData-1,st,playbackSpeed)
            sPoints.Text=string.format("📍%d  |  🕐%.1fs  |  %s",#recordedData,elapsed,st)
            refreshPlayUI()
        end
        end)
    end)
    refreshPlayUI()
end

-- WALK TO NEAREST THEN PLAY
local function stopWalkToTrack()
    if walkToTrackConn then walkToTrackConn:Disconnect(); walkToTrackConn=nil end
    isWalkingToTrack=false; refreshPlayUI()
end

local function walkToNearestThenPlay()
    if #recordedData<2 then notify("⚠ Error","Butuh ≥ 2 titik",2); return end
    local ch=player.Character; if not ch then return end
    humanoid=ch:FindFirstChildOfClass("Humanoid"); rootPart=ch:FindFirstChild("HumanoidRootPart")
    if not humanoid or not rootPart then return end
    local nearIdx=findNearestIndex(); local nearPos=recordedData[nearIdx].Position
    local dist=(nearPos-rootPart.Position).Magnitude
    if dist<5 then
        notify("▶ Play Terdekat","Langsung mulai dari titik "..nearIdx,2); startPlayback(true); return
    end
    notify("🏃 Menuju Track","Jalan ke titik "..nearIdx.." ("..math.floor(dist).."s)",3)
    isWalkingToTrack=true; updateStatus("🏃 Menuju Track...","Titik "..nearIdx.." — "..math.floor(dist).." studs",Color3.fromRGB(255,200,0),Color3.fromRGB(200,160,0),true)
    refreshPlayUI()
    humanoid:MoveTo(nearPos)
    local timeout=0
    walkToTrackConn=RunService.Heartbeat:Connect(function(dt)
        if not isWalkingToTrack then return end
        timeout=timeout+dt
        local ch2=player.Character; local rp2=ch2 and ch2:FindFirstChild("HumanoidRootPart")
        if not rp2 then stopWalkToTrack(); return end
        local d2=(nearPos-rp2.Position).Magnitude
        if d2<4.5 or timeout>=15 then
            stopWalkToTrack()
            task.wait(0.2)
            notify("▶ Sampai!","Mulai playback dari titik "..nearIdx,2)
            startPlayback(true)
        elseif math.floor(timeout*10)%50==0 then
            -- refresh MoveTo agar tidak expire
            local hm2=ch2 and ch2:FindFirstChildOfClass("Humanoid"); if hm2 then hm2:MoveTo(nearPos) end
        end
    end)
end

        local urlRaw=wrBox.Text:gsub("%s+","")

                wrLoadBtn.Text="🌐 Load dari Web"; wrLoadBtn.Active=true; wrAppendBtn.Active=true; return
            end
            local ok2,loaded=pcall(function()
                -- Coba decode JSON → deserialize sebagai route data
                local hs=game:GetService("HttpService")
                local decoded=hs:JSONDecode(raw)
                -- Format 1: {points=[...]} wrapper dari server
                if decoded.points then return deserializeData(hs:JSONEncode(decoded.points)) end
                -- Format 2: {data="<json string>"} wrapper
                if decoded.data and type(decoded.data)=="string" then return deserializeData(decoded.data) end
                -- Format 3: langsung array
                if type(decoded)=="table" and decoded[1] and decoded[1].px then return deserializeData(raw) end
                -- Format 4: mungkin raw JSON array langsung
                return deserializeData(raw)
            end)
            if ok2 and loaded and #loaded>=2 then
                if append then
                    for _,d in ipairs(loaded) do table.insert(recordedData,d) end

                notify("⚠ Web Route","Gagal parse data: "..errMsg:sub(1,40),4)
            end
            wrLoadBtn.Text="🌐 Load dari Web"; wrLoadBtn.Active=true; wrAppendBtn.Active=true
        end)

end)
